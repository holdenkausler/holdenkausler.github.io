                    $('#answer'+i).click(function(){
                        reveal(i, currentQuestion);
                    });
                }
            }
    
            function reveal(num, currentQuestion2) {
                if(disableButtons){
                    return;
                }
                var currentAnswerNumber;
                if(currentQuestion2.answer[0].isCorrect){
                    currentAnswerNumber=0;
                }else if(currentQuestion2.answer[1].isCorrect){
                    currentAnswerNumber=1;
                }else if(currentQuestion2.answer[2].isCorrect){
                    currentAnswerNumber=2;
                }else if(currentQuestion2.answer[3].isCorrect){
                    currentAnswerNumber=3;
                }else{
                    console.log('None of the answers are recognized as correct, idiot');
                }
    
                if(num==currentAnswerNumber){
                    correctSoFar++;
                    console.log('correct Answer Chosen');
                    $('#reveal').show();
                }
    
            }
    })
